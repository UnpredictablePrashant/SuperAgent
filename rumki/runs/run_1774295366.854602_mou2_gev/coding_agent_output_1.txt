class RepositoryScanner:
    def __init__(self, root: Path) -> None:
        self.root = root.resolve()
        self.files: List[Path] = []
        self._scan()

    def _scan(self) -> None:
        for dirpath, dirnames, filenames in os.walk(self.root):
            dirnames[:] = [d for d in dirnames if d not in EXCLUDED_DIRS and not d.startswith(".eggs")]
            base = Path(dirpath)
            for filename in filenames:
                p = base / filename
                self.files.append(p)

    def rel(self, p: Path) -> str:
        return str(p.relative_to(self.root)).replace("\\", "/")

    def exists(self, rel_path: str) -> bool:
        return (self.root / rel_path).exists()

    def find(self, name: str) -> List[Path]:
        n = name.lower()
        return [p for p in self.files if p.name.lower() == n]

    def glob_rel(self, pattern: str) -> List[Path]:
        return list(self.root.glob(pattern))

    def list_relpaths(self) -> List[str]:
        return [self.rel(p) for p in self.files]

    def top_level(self) -> Tuple[List[str], List[str]]:
        dirs, files = [], []
        for p in self.root.iterdir():
            if p.name in EXCLUDED_DIRS:
                continue
            if p.is_dir():
                dirs.append(p.name)
            elif p.is_file():
                files.append(p.name)
        return sorted(dirs), sorted(files)

    def language_stats(self) -> List[Tuple[str, int]]:
        c: Counter[str] = Counter()
        for p in self.files:
            ext = p.suffix.lower()
            lang = LANGUAGE_BY_EXTENSION.get(ext)
            if lang:
                c[lang] += 1
            elif p.name == "Dockerfile":
                c["Dockerfile"] += 1
        return sorted(c.items(), key=lambda x: x[1], reverse=True)

    def read_text(self, p: Path, max_bytes: int = 1_000_000) -> Optional[str]:
        try:
            if p.stat().st_size > max_bytes:
                return None
            return p.read_text(encoding="utf-8", errors="ignore")
        except Exception:
            return None

    def is_text_file(self, p: Path) -> bool:
        if p.name in {"Dockerfile", "Jenkinsfile", "Makefile"}:
            return True
        return p.suffix.lower() in TEXT_EXTENSIONS

    def _scan_for_secrets(self) -> List[str]:
        hits: List[str] = []
        for p in self.scanner.files:
            if not self.scanner.is_text_file(p):
                continue
            text = self.scanner.read_text(p, max_bytes=500_000)
            if not text:
                continue
            lines = text.splitlines()
            for i, line in enumerate(lines, start=1):
                if len(line) > 5000:
                    continue
                for pattern in SECRET_PATTERNS:
                    if pattern.search(line):
                        hits.append(f"{self.scanner.rel(p)}:{i}")
                        if len(hits) >= 20:
                            return hits
                        break
        return hits

    def _build_prioritized_plan(self) -> List[Dict[str, Any]]:
        buckets: Dict[str, List[Finding]] = defaultdict(list)
        for f in self.findings:
            buckets[f.priority].append(f)

        phases = [("P0", "Immediate (0-2 days)"), ("P1", "Short-term (this sprint)"), ("P2", "Near-term (next sprint)"), ("P3", "Backlog")]
        plan: List[Dict[str, Any]] = []
        for priority, window in phases:
            items = buckets.get(priority, [])
            if not items:
                continue
            plan.append(
                {
                    "priority": priority,
                    "timeline": window,
                    "count": len(items),
                    "actions": [
                        {
                            "id": i.id,
                            "title": i.title,
                            "category": i.category,
                            "delegate_to": i.delegate_to,
                            "implementation_needed": i.implementation_needed,
                        }
                        for i in items
                    ],
                }
            )
        return plan

    def _build_delegation_tasks(self) -> List[Dict[str, Any]]:
        tasks = []
        for f in self.findings:
            if not f.implementation_needed:
                continue
            tasks.append(
                {
                    "task_id": f.id,
                    "owner": f.delegate_to,
                    "task": f.title,
                    "acceptance_criteria": f.recommendation,
                    "priority": f.priority,
                }
            )
        return tasks


def render_markdown(report: Dict[str, Any]) -> str:
    s = report["summary"]
    findings = report["findings"]
    plan = report["prioritized_plan"]
    delegated = report["delegated_followups"]

    lines: List[str] = []
    lines.append("# Production Readiness Assessment")
    lines.append("")
    lines.append("## Repository Inspection")
    lines.append(f"- Path: `{s['path']}`")
    lines.append(f"- Total files scanned: **{s['total_files']}**")

    lines.append(f"- Top-level files: {', '.join(s['top_level_files']) if s['top_level_files'] else '(none)'}")

    lines.append(f"- Detected languages: {lang_desc}")
    lines.append("")
    lines.append("## Key File Presence")
    for k, v in s["key_files_present"].items():
        lines.append(f"- {k}: {'✅' if v else '❌'}")

    lines.append("")
    lines.append("## Prioritized Remediation Plan")
    if not plan:
        lines.append("- No critical gaps detected by heuristic checks.")
    else:
        for phase in plan:
            lines.append(f"### {phase['priority']} — {phase['timeline']}")
            for action in phase["actions"]:
                impl = "implementation required" if action["implementation_needed"] else "policy/process check"
                lines.append(
                    f"- [{action['id']}] {action['title']} ({action['category']}) — Owner: {action['delegate_to']} ({impl})"
                )

    lines.append("")
    lines.append("## Detailed Findings")
    if not findings:
        lines.append("- No findings.")
    else:
        for f in findings:
            lines.append(f"### {f['id']} — {f['title']} ({f['priority']})")
            lines.append(f"- Category: {f['category']}")
            lines.append(f"- Evidence: {f['evidence']}")
            lines.append(f"- Recommended fix: {f['recommendation']}")
            lines.append(f"- Delegate to: {f['delegate_to']}")
            lines.append(f"- Effort/Impact: {f['effort']}/{f['impact']}")
            lines.append(f"- Implementation needed: {'Yes' if f['implementation_needed'] else 'No'}")
            lines.append("")

    lines.append("## Delegated Follow-up Tasks")
    if not delegated:
        lines.append("- None.")
    else:
        for d in delegated:
            lines.append(
                f"- [{d['priority']}] {d['task_id']} -> {d['owner']}: {d['task']} | Acceptance: {d['acceptance_criteria']}"
            )

    return "\n".join(lines).strip() + "\n"


def parse_args(argv: Optional[Sequence[str]] = None) -> argparse.Namespace:
    parser = argparse.ArgumentParser(
        description="Assess repository production-readiness and generate a prioritized remediation plan."
    )
    parser.add_argument("path", nargs="?", default=".", help="Repository path (default: current directory)")
    parser.add_argument(
        "--format",
        choices=["markdown", "json"],
        default="markdown",
        help="Output format",
    )
    parser.add_argument("--out", default="", help="Optional output file path; prints to stdout if omitted")
    return parser.parse_args(argv)


def main(argv: Optional[Sequence[str]] = None) -> int:
    args = parse_args(argv)
    root = Path(args.path).resolve()
    if not root.exists() or not root.is_dir():
        print(f"Error: path does not exist or is not a directory: {root}", file=sys.stderr)
        return 2

    scanner = RepositoryScanner(root)
    assessor = ProductionReadinessAssessor(scanner)
    report = assessor.assess()

    if args.format == "json":
        output = json.dumps(report, indent=2)
    else:
        output = render_markdown(report)

    if args.out:
        out_path = Path(args.out)
        out_path.parent.mkdir(parents=True, exist_ok=True)
        out_path.write_text(output, encoding="utf-8")
    else:
        sys.stdout.write(output)

    return 0


if __name__ == "__main__":
    raise SystemExit(main())
